# Makefile rule to compile a C program
# Assumes the source file is named 'main.c' and the output executable is named 'program'

CC = gcc
CFLAGS = -Wall -g # Add warning flags and debug info
TARGET = program
SOURCE = main.c

$(TARGET): $(SOURCE)
	$(CC) $(CFLAGS) -o $(TARGET) $(SOURCE)

clean:
	rm -f $(TARGET)

.PHONY: clean
